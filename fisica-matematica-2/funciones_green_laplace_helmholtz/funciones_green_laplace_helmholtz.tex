\problema{0}{Recordatorio: Laplaciano en coordenadas polares y esfÃĐricas}

Para una funciÃģn escalar $F=F(\rho,\varphi)$, el operador Laplaciano en coordenadas polares se escribe como
\begin{equation*}
\nabla^2F=
\frac{1}{\rho}\frac{\partial\ }{\partial\rho}
\left(\rho\frac{\partial F}{\partial\rho}\right)
+\frac{1}{\rho^2}\frac{\partial^2F}{\partial\varphi^2}.
\end{equation*}
Si $F$ es radial, es decir $F=F(\rho)$, la parte angular se anula y queda
\begin{equation*}
\nabla^2F=
\frac{1}{\rho}\frac{\dd\ }{\dd\rho}
\left(\rho\frac{\dd F}{\dd\rho}\right).
\end{equation*}

Para una funciÃģn escalar $F=F(r,\theta,\varphi)$, el operador Laplaciano en coordenadas esfÃĐricas se escribe como
\begin{equation*}
\nabla^2F=
\frac{1}{r^2}\frac{\partial\ }{\partial r}
\left(r^2\frac{\partial F}{\partial r}\right)
+\frac{1}{r^2\sen\theta}\frac{\partial\ }{\partial\theta}
\left(\sen\theta\frac{\partial F}{\partial\theta}\right)
+\frac{1}{r^2\sen^2\theta}\frac{\partial^2F}{\partial\varphi^2}.
\end{equation*}
Si $F$ es radial, es decir $F=F(r)$, la parte angular se anula y queda
\begin{equation*}
\nabla^2F=
\frac{1}{r^2}\frac{\dd\ }{\dd r}
\left(r^2\frac{\dd F}{\dd r}\right).
\end{equation*}

La funciÃģn de Green definida por la ecuaciÃģn de Green no es Ãšnica. Si $G_1(\vec{x},\vec{x}')$ es soluciÃģn, entonces $G_2(\vec{x},\vec{x}'):=G_1(\vec{x},\vec{x}')+H(\vec{x},\vec{x}')$ tambiÃĐn es soluciÃģn, si $H(\vec{x},\vec{x}')$ es soluciÃģn del problema homogÃĐneo asociado,
\begin{equation}
\hat{L}H(\vec{x},\vec{x}')=0.
\end{equation}
Este hecho permite considerar soluciones particulares simples como base para otras funciones de Green, que pueden obtenerse agregando una soluciÃģn de la ecuaciÃģn homogÃĐnea de modo que la funciÃģn resultante satisfaga las condiciones de borde que simplifiquen el problema.

\problema{1}{Operador Laplaciano en $D=3$}

\textbf{SoluciÃģn:} En este caso $p(\vec{x})=1$ y $q(\vec{x})=0$. En FÃ­sica, usualmente se busca la funciÃģn de Green que ``respete la homogeneidad e isotropÃ­a del espacio''. La primera condiciÃģn implica que $G$ depende de $\vec{x}$ y $\vec{x}'$ sÃģlo a travÃĐs de su diferencia $\vec{x}-\vec{x}'$. La segunda condiciÃģn implica que $G$ sÃģlo depende del mÃģdulo de $\vec{x}-\vec{x}'$. Entonces
\begin{equation}
G(\vec{x},\vec{x}')=G(|\vec{x}-\vec{x}'|).
\end{equation}

En coordenadas esfÃĐricas centradas en $\vec{x}'$, escribimos $r:=|\vec{x}-\vec{x}'|$. Como $G=G(r)$, se tiene
\begin{equation}\label{n2Gd}
\nabla^2G=\frac{1}{r^2}\frac{\dd\ }{\dd r}\left(r^2\frac{\dd G}{\dd r}\right)=\delta^{(3)}(r).
\end{equation}
Por lo tanto, fuera del origen,
\begin{equation}\label{lap3_hom}
\frac{1}{r^2}\frac{\dd\ }{\dd r}\left(r^2\frac{\dd G}{\dd r}\right)=0,\qquad r\neq 0.
\end{equation}
La soluciÃģn para $r\neq 0$ se encuentra integrando directamente \eqref{lap3_hom}. En efecto,
\begin{align*}
\frac{\dd\ }{\dd r}\left(r^2\frac{\dd G}{\dd r}\right)&=0,\\
r^2\frac{\dd G}{\dd r}&=C_1,\\
\frac{\dd G}{\dd r}&=\frac{C_1}{r^2}.
\end{align*}
Integrando una vez mÃĄs,
\begin{equation*}
G(r)=C_2-\frac{C_1}{r}.
\end{equation*}
Renombrando constantes, obtenemos
\begin{equation}\label{lap3_general}
G(r)=\alpha+\frac{\beta}{r}.
\end{equation}

Por otro lado, la condiciÃģn \eqref{n2Gd} implica, usando el teorema de Gauss, que
\begin{equation}\label{gauss_lap3}
\int_{\partial V}\vec\nabla G\cdot \dd\vec{S}
=\int_{\partial V}\frac{\dd G}{\dd r}r^2\dd\Omega=1.
\end{equation}
Desde \eqref{lap3_general},
\begin{equation*}
\frac{\dd G}{\dd r}=-\frac{\beta}{r^2}.
\end{equation*}
Luego, sobre una esfera de radio $R$,
\begin{align*}
\int_{\partial V}\frac{\dd G}{\dd r}r^2\dd\Omega
&=\int_{\partial V}\left(-\frac{\beta}{R^2}\right)R^2\dd\Omega\\
&=-\beta\int\dd\Omega\\
&=-4\pi\beta.
\end{align*}
La condiciÃģn \eqref{gauss_lap3} requiere entonces que $-4\pi\beta=1$, o sea
\begin{equation*}
\beta=-\frac{1}{4\pi}.
\end{equation*}
Finalmente, en la mayorÃ­a de los problemas es conveniente elegir una funciÃģn de Green que se anule para distancias muy grandes, es decir, tal que $\lim_{r\to\infty}G(r)=0$. Esta condiciÃģn impone que $\alpha=0$ y por lo tanto
\begin{equation}
\boxed{G(\vec{x}-\vec{x}')=-\frac{1}{4\pi}\frac{1}{|\vec{x}-\vec{x}'|}.}
\end{equation}

\problema{2}{Operador Laplaciano en $D=2$}

\textbf{SoluciÃģn:} En coordenadas polares $(\rho,\varphi)$ centradas en $\vec{x}'$, escribimos $\rho:=|\vec{x}-\vec{x}'|$. Como $G=G(\rho)$,
\begin{equation}\label{lap2_delta}
\nabla^2G=\frac{1}{\rho}\frac{\dd\ }{\dd\rho}\left(\rho\frac{\dd G}{\dd\rho}\right)=\delta^{(2)}(\rho).
\end{equation}
Integrando la ecuaciÃģn para $\rho\neq0$, es decir,
\begin{equation}\label{lap2_hom}
\frac{1}{\rho}\frac{\dd\ }{\dd\rho}\left(\rho\frac{\dd G}{\dd\rho}\right)=0,
\end{equation}
encontramos
\begin{align*}
\frac{\dd\ }{\dd\rho}\left(\rho\frac{\dd G}{\dd\rho}\right)&=0,\\
\rho\frac{\dd G}{\dd\rho}&=C_1,\\
\frac{\dd G}{\dd\rho}&=\frac{C_1}{\rho},\\
G(\rho)&=C_2+C_1\ln\rho.
\end{align*}
Renombrando constantes,
\begin{equation}\label{lap2_general}
G(\rho)=\alpha+\beta\ln\rho.
\end{equation}

Nuevamente, el teorema de Gauss, en su versiÃģn 2D, implica
\begin{equation}\label{gauss_lap2}
\int_{\partial S}\vec\nabla G\cdot \dd\vec{S}
=\oint\frac{\dd G}{\dd\rho}\rho\,\dd\varphi=2\pi\beta=1.
\end{equation}
La Ãšltima igualdad se obtiene directamente desde \eqref{lap2_general}, pues
\begin{equation*}
\frac{\dd G}{\dd\rho}=\frac{\beta}{\rho},
\end{equation*}
y por lo tanto
\begin{align*}
\oint\frac{\dd G}{\dd\rho}\rho\,\dd\varphi
&=\oint\frac{\beta}{\rho}\rho\,\dd\varphi\\
&=\beta\int_0^{2\pi}\dd\varphi\\
&=2\pi\beta.
\end{align*}
AsÃ­, $2\pi\beta=1$, de donde $\beta=1/(2\pi)$. De este modo, eliminando el tÃĐrmino constante, que es una soluciÃģn de la ecuaciÃģn homogÃĐnea, encontramos
\begin{equation}
\boxed{G(\vec{x}-\vec{x}')=\frac{1}{2\pi}\ln{|\vec{x}-\vec{x}'|}.}
\end{equation}

\problema{3}{Operador de Helmholtz en $D=3$}

\textbf{SoluciÃģn:} En el caso del operador de Helmholtz $\hat{L}=\nabla^2+k^2$, corresponde tomar $p(\vec{x})=1$ y $q(\vec{x})=k^2$. Buscaremos la funciÃģn de Green de la forma
\begin{equation*}
G(\vec{x},\vec{x}')=G(|\vec{x}-\vec{x}'|).
\end{equation*}

En coordenadas esfÃĐricas centradas en $\vec{x}'$, con $r:=|\vec{x}-\vec{x}'|$,
\begin{equation}\label{HGd}
(\nabla^2+k^2)G
=\frac{1}{r^2}\frac{\dd\ }{\dd r}\left(r^2\frac{\dd G}{\dd r}\right)+k^2G
=\delta^{(3)}(r),
\end{equation}
por lo tanto, para $r\neq0$,
\begin{equation}\label{HG0}
\frac{1}{r^2}\frac{\dd\ }{\dd r}\left(r^2\frac{\dd G}{\dd r}\right)+k^2G=0,\qquad r\neq0.
\end{equation}
Para solucionar \eqref{HG0} realizamos el cambio de variable
\begin{equation*}
G(r)=\frac{u(r)}{r}.
\end{equation*}
Entonces
\begin{align*}
\frac{\dd G}{\dd r}
&=\frac{\dd\ }{\dd r}\left(\frac{u}{r}\right)\\
&=\frac{u'}{r}-\frac{u}{r^2}\\
&=\frac{ru'-u}{r^2},
\end{align*}
y por lo tanto
\begin{equation*}
r^2\frac{\dd G}{\dd r}=ru'-u.
\end{equation*}
Derivando,
\begin{align*}
\frac{\dd\ }{\dd r}\left(r^2\frac{\dd G}{\dd r}\right)
&=\frac{\dd\ }{\dd r}(ru'-u)\\
&=u'+ru''-u'\\
&=ru''.
\end{align*}
Con esto, \eqref{HG0} se reduce a
\begin{equation*}
\frac{1}{r^2}(ru'')+k^2\frac{u}{r}=0,
\end{equation*}
es decir,
\begin{equation*}
u''+k^2u=0.
\end{equation*}
Por lo tanto, las soluciones son de la forma
\begin{equation}\label{GH3d}
G(r)=\alpha\frac{\e^{\ii kr}}{r}+\beta\frac{\e^{-\ii kr}}{r}.
\end{equation}

Con esta soluciÃģn, vÃĄlida para $r\neq0$, retornamos a la ecuaciÃģn \eqref{HGd} que, luego de integrar sobre una esfera de radio $R$ centrada en $r=0$ y usando el teorema de Gauss, implica que
\begin{align}
1 &= \oint_{\partial V}\vec\nabla G\cdot \dd\vec{S}+k^2\int_VG\,\dd V \label{helm_int1}\\
&= \oint_{\partial V}\frac{\dd G}{\dd r}r^2\dd\Omega+k^2\int_VGr^2\,\dd r\dd\Omega \label{helm_int2}\\
&= 4\pi G'R^2+4\pi k^2\int_0^R Gr^2\,\dd r \label{helm_int3}\\
&= 4\pi\left[\alpha\left(\ii kR-1\right)\e^{\ii kR}-\beta\left(\ii kR+1\right)\e^{-\ii kR}\right]
+4\pi k^2\int_0^R \left[\alpha\e^{\ii kr}+\beta\e^{-\ii kr}\right]r\,\dd r \label{helm_int4}\\
&= -4\pi(\alpha+\beta). \label{helm_int5}
\end{align}
En \eqref{helm_int4} se usÃģ que, desde \eqref{GH3d},
\begin{align*}
\frac{\dd G}{\dd r}
&=\alpha\frac{\dd\ }{\dd r}\left(\frac{\e^{\ii kr}}{r}\right)
+\beta\frac{\dd\ }{\dd r}\left(\frac{\e^{-\ii kr}}{r}\right)\\
&=\alpha\frac{(\ii kr-1)\e^{\ii kr}}{r^2}
-\beta\frac{(\ii kr+1)\e^{-\ii kr}}{r^2}.
\end{align*}
AdemÃĄs,
\begin{align*}
k^2\int_0^R \e^{\ii kr}r\,\dd r &= \e^{\ii kR}(1-\ii kR)-1,\\
k^2\int_0^R \e^{-\ii kr}r\,\dd r &= \e^{-\ii kR}(1+\ii kR)-1.
\end{align*}
Por lo tanto, el tÃĐrmino de borde mÃĄs el tÃĐrmino de volumen dentro de \eqref{helm_int4} queda
\begin{align*}
&\alpha(\ii kR-1)\e^{\ii kR}-\beta(\ii kR+1)\e^{-\ii kR}\\
&\qquad +\alpha\left[\e^{\ii kR}(1-\ii kR)-1\right]
+\beta\left[\e^{-\ii kR}(1+\ii kR)-1\right]\\
&=-(\alpha+\beta),
\end{align*}
lo cual da el paso final de \eqref{helm_int4} a \eqref{helm_int5}.

De esta forma encontramos la condiciÃģn
\begin{equation}\label{cond_ab}
\alpha+\beta=-\frac{1}{4\pi},
\end{equation}
que permite escribir la soluciÃģn \eqref{GH3d} como
\begin{equation}\label{GH3d2}
G(r)=-\frac{1}{4\pi}\frac{\e^{\ii kr}}{r}-2\ii\beta\frac{\sen(kr)}{r}.
\end{equation}
En efecto, usando \eqref{cond_ab}, se tiene $\alpha=-1/(4\pi)-\beta$, y entonces
\begin{align*}
G(r)
&=\left(-\frac{1}{4\pi}-\beta\right)\frac{\e^{\ii kr}}{r}+\beta\frac{\e^{-\ii kr}}{r}\\
&=-\frac{1}{4\pi}\frac{\e^{\ii kr}}{r}+\beta\frac{\e^{-\ii kr}-\e^{\ii kr}}{r}\\
&=-\frac{1}{4\pi}\frac{\e^{\ii kr}}{r}-2\ii\beta\frac{\sen(kr)}{r}.
\end{align*}
El segundo tÃĐrmino es una soluciÃģn de la ecuaciÃģn de Helmholtz \textit{homogÃĐnea}; es proporcional a la funciÃģn esfÃĐrica de Bessel $j_0(kr)$. Por lo tanto, es posible elegir
\begin{equation}\label{GH3d3}
\boxed{G^{+}(\vec{x},\vec{x}')=-\frac{1}{4\pi}\frac{\e^{\ii k|\vec{x}-\vec{x}'|}}{|\vec{x}-\vec{x}'|},}
\end{equation}
que corresponde a la elecciÃģn $\beta=0$. AnÃĄlogamente, podemos considerar la funciÃģn de Green
\begin{equation}\label{GH3d4}
G^{-}(\vec{x},\vec{x}')=-\frac{1}{4\pi}\frac{\e^{-\ii k|\vec{x}-\vec{x}'|}}{|\vec{x}-\vec{x}'|},
\end{equation}
que se encuentra en el caso en que $\beta=-1/4\pi$.
